----------Условие----------
Учитывая строку s, содержащую только символы '(', ')', '{', и , определите '}', допустима ли входная строка.'['']'
Входная строка действительна, если:
Открытые скобки должны быть закрыты однотипными скобками.
Открытые скобки должны быть закрыты в правильном порядке.
Каждой закрывающей скобке соответствует открытая скобка того же типа.

Пример 1:

Ввод: s = "()"
 Вывод: правда
Пример 2:

Ввод: s = "()[]{}"
 Вывод: правда
Пример 3:

Ввод: s = "(]"
 Вывод: ложь

----------Решение----------
func isValid(s string) bool {
    stack := []rune{} // стек для хранения открытых скобок
    mapping := map[rune]rune{')': '(', '}': '{', ']': '['} // маппинг открытых и закрытых скобок

    for _, char := range s {
        // если текущий символ - открытая скобка, добавляем ее в стек
        if char == '(' || char == '{' || char == '[' {
            stack = append(stack, char)
        } else if len(stack) == 0 || stack[len(stack)-1] != mapping[char] {
            // если текущий символ - закрытая скобка, но нет открытой скобки для закрытия, или закрытая скобка не соответствует
		последней открытой, то возвращаем false
            return false
        } else {
            // если текущая закрывающая скобка соответствует последней открытой, удаляем последнюю открытую скобку из стека
            stack = stack[:len(stack)-1]
        }
    }

    // если стек пуст, значит все скобки были закрыты, и строка валидна
    return len(stack) == 0
}
